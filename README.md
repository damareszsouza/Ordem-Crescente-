# Ordem-Crescente-
Exercício de logica de programação 1

Ordem crescente
Gabarito do Exercício 3
programa

{

    funcao inicio(){

        real valor1, valor2

        escreva ("Informe a nota da valor 1: ")

        leia(valor1)

        escreva ("Informe a nota da valor 2: ")

        leia(valor2)

        se(valor1 < valor2) {

            escreva ("A ordem é: ", valor1, ", ", valor2)

        }

        senao{

            escreva ("A ordem é: ", valor2, ", ", valor1)

        }

    }

}
